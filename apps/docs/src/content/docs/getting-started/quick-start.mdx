## Basic Setup

After [installing](/getting-started/installation/) FlowDrop, import the styles and mount the editor:

```svelte
<script lang="ts">
  import '@d34dman/flowdrop/styles';
  import { mountFlowDropApp } from '@d34dman/flowdrop';
  import { onMount } from 'svelte';

  let container: HTMLDivElement;

  onMount(() => {
    const app = mountFlowDropApp(container, {
      // Your configuration here
    });

    return () => app.destroy();
  });
</script>

<div bind:this={container} style="width: 100%; height: 600px;" />
```
